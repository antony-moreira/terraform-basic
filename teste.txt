from lxml import etree

# Carregar o XML
tree = etree.parse('seu_arquivo.xml')
root = tree.getroot()

# Definir o namespace necessário
namespace = {
    'xsi': 'http://www.w3.org/2001/XMLSchema-instance',
    'job': 'http://com.teste/teste/teste-xmlnsdaaw'
}

# Encontrar o elemento que contém xsi:type
job_step_element = root.find(".//job:job-step", namespaces=namespace)

# Extrair o valor de xsi:type
xsi_type = job_step_element.get(f'{{{namespace["xsi"]}}}type')

# Imprimir o valor de xsi:type
print(f'Valor de xsi:type: {xsi_type}')
